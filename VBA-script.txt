Sub stock_info()

    'Set variables
    Dim ticker As String
    
    Dim yearly_change As Double
    
    Dim per_change As Double
    
    Dim total_volume As Double
    total_volume = 0
    
    'Keep track of location of tickers in summary table
    Dim summary_table_row As Integer
    summary_table_row = 2
    
    'Add headers for the summary table
    Range("I1").Value = "Ticker"
    Range("J1").Value = "Yearly Change"
    Range("K1").Value = "Percentage Change"
    Range("L1").Value = "Total Stock Volume"
    
    'Find the last row and declare it as a variable
    last_row = Cells(Rows.Count, 1).End(xlUp).Row

    'Loop through the stocks
    For i = 2 To last_row
    
        'Create if statement to check if we are within the same ticker
        If Cells(i + 1, 1).Value <> Cells(i, 1).Value Then
        
        'If ticker is not the same;
        'Set ticker
        ticker = Cells(i, 1).Value
        
        'Print the ticker in the summary table
        Range("I" & summary_table_row).Value = ticker
        
        'Declare open and closing values
        closing_price = Cells(i, 6).Value
        opening_price = Cells(i - 1, 3).Value
    
        'Add yearly change
        yearly_change = closing_price - opening_price
        
        'Print yearly change in summary table
        Range("J" & summary_table_row).Value = yearly_change
        
        'Add percentage change
        per_change = (closing_price - opening_price) / opening_price
        
        'Print percentage change in summary table
        Range("K" & summary_table_row).Value = Format(per_change, "percent")
    
        'Add total stock volume
        total_volume = total_volume + Cells(i, 7).Value
        
        'Print total stock volume in summary table
        Range("L" & summary_table_row).Value = total_volume
        
        'Reset total stock volume
        total_volume = 0
        
        'Add one to the summary table row
        summary_table_row = summary_table_row + 1
        
        'Close if statement
        End If
    
        'Create an if statement; if positive yearly chanage then colour cell green
        If Cells(i, 10).Value > 0 Then
        Cells(i, 10).Interior.ColorIndex = 4
